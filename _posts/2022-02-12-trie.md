#include<bits/stdc++.h>
using namespace std;
int n,m;
int edge[500010][26],tot;
int ex[500010];
void insert(char *s,int ls)
{
	int p=0;
	for(int i=1;i<=ls;++i)
	{
		int c=s[i]-'a';
		if(!edge[p][c]) edge[p][c]=++tot;
		p=edge[p][c];
	}
	++ex[p];
}
bool find(char *s,int ls)
{
	int p=0;
	for(int i=1;i<=ls;++i)
	{
		int c=s[i]-'a';
		if(!edge[p][c]) return 0;
		p=edge[p][c];
	}
	if(!ex[p]) return 0;
	return 1;
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	{
		char na[110]="";
		scanf("%s",na+1);
		int len=strlen(na+1);
		insert(na,len);
	}
	scanf("%d",&m);
	for(int i=1;i<=m;++i)
	{
		char na[110]="";
		scanf("%s",na+1);
		int len=strlen(na+1);
		printf("%d\n",find(na,len));
	}
	return 0;
}
